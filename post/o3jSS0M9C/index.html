反爬虫机制2:代理IP爬去网站

2019-07-28 
proxylist = [
    {"http":"47.107.186.18:8118"},
    {"http":"101.132.164.113:8118"},
    {"http":"121.232.194.196:9000"},
    {"http":"112.111.97.101:9000"},
    {"http":"175.44.151.198:9000"}
]

# 随机选取代理IP
proxy = random.choice(proxylist)
print(proxy)

# 构建代理处理器对象
proxyHandler = request.ProxyHandler(proxy)

# 创建自定义opener
opener = request.build_opener(proxyHandler)

# 创建自定义请求对象
request = request.Request(url)

response = opener.open(request)
